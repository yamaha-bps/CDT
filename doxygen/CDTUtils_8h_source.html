370 template <>
371 struct hash<CDT::Edge>
372 {
374  std::size_t operator()(const CDT::Edge& e) const
375  {
376  return hashEdge(e);
377  }
378 
379 private:
380  static void hashCombine(std::size_t& seed, const CDT::VertInd& key)
381  {
382 #ifdef CDT_CXX11_IS_SUPPORTED
383  typedef std::hash<CDT::VertInd> Hasher;
384 #else
385  typedef boost::hash<CDT::VertInd> Hasher;
386 #endif
387  seed ^= Hasher()(key) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
388  }
389  static std::size_t hashEdge(const CDT::Edge& e)
390  {
391  const std::pair<CDT::VertInd, CDT::VertInd>& vv = e.verts();
392  std::size_t seed1(0);
393  hashCombine(seed1, vv.first);
394  hashCombine(seed1, vv.second);
395  std::size_t seed2(0);
396  hashCombine(seed2, vv.second);
397  hashCombine(seed2, vv.first);
398  return std::min(seed1, seed2);
399  }
400 };
