68 template <typename T>
69 void Triangulation<T>::eraseDummies()
70 {
71  if(m_dummyTris.empty())
72  return;
73  const TriIndUSet dummySet(m_dummyTris.begin(), m_dummyTris.end());
74  TriIndUMap triIndMap;
75  triIndMap[noNeighbor] = noNeighbor;
76  for(TriInd iT(0), iTnew(0); iT < TriInd(triangles.size()); ++iT)
77  {
78  if(dummySet.count(iT))
79  continue;
80  triIndMap[iT] = iTnew;
81  triangles[iTnew] = triangles[iT];
82  iTnew++;
83  }
84  triangles.erase(triangles.end() - dummySet.size(), triangles.end());
85 
86  // remap adjacent triangle indices for vertices
87  typedef typename VertexVec::iterator VertIt;
88  for(VertIt v = vertices.begin(); v != vertices.end(); ++v)
89  {
90  TriIndVec& vTris = v->triangles;
91  for(TriIndVec::iterator iT = vTris.begin(); iT != vTris.end(); ++iT)
92  *iT = triIndMap[*iT];
93  }
94  // remap neighbor indices for triangles
95  for(TriangleVec::iterator t = triangles.begin(); t != triangles.end(); ++t)
96  {
97  NeighborsArr3& nn = t->neighbors;
98  for(NeighborsArr3::iterator iN = nn.begin(); iN != nn.end(); ++iN)
99  *iN = triIndMap[*iN];
100  }
101  // clear dummy triangles
102  m_dummyTris = std::vector<TriInd>();
103 }

178 template <typename T>
179 TriIndUSet Triangulation<T>::growToBoundary(std::stack<TriInd> seeds) const
180 {
181  TriIndUSet traversed;
182  while(!seeds.empty())
183  {
184  const TriInd iT = seeds.top();
185  seeds.pop();
186  traversed.insert(iT);
187  const Triangle& t = triangles[iT];
188  for(Index i(0); i < Index(3); ++i)
189  {
190  const Edge opEdge(t.vertices[ccw(i)], t.vertices[cw(i)]);
191  if(fixedEdges.count(opEdge))

195  seeds.push(iN);
196  }
197  }

258 template <typename T>
259 void Triangulation<T>::insertEdge(Edge edge)
260 {
261  const VertInd iA = edge.v1();
262  VertInd iB = edge.v2();
263  if(iA == iB) // edge connects a vertex to itself
264  return;
265  const Vertex<T>& a = vertices[iA];
266  const Vertex<T>& b = vertices[iB];
267  if(verticesShareEdge(a, b))
268  {
269  fixedEdges.insert(Edge(iA, iB));
270  return;
271  }
272  TriInd iT;
273  VertInd iVleft, iVright;
274  tie(iT, iVleft, iVright) =
275  intersectedTriangle(iA, a.triangles, a.pos, b.pos);
276  // if one of the triangle vertices is on the edge, move edge start
277  if(iT == noNeighbor)
278  {
279  fixedEdges.insert(Edge(iA, iVleft));
280  return insertEdge(Edge(iVleft, iB));
281  }
282  std::vector<TriInd> intersected(1, iT);
283  std::vector<VertInd> ptsLeft(1, iVleft);
284  std::vector<VertInd> ptsRight(1, iVright);
285  VertInd iV = iA;
286  Triangle t = triangles[iT];
287  const VerticesArr3& tverts = t.vertices;
288  while(std::find(tverts.begin(), tverts.end(), iB) == tverts.end())
289  {
290  const TriInd iTopo = opposedTriangle(t, iV);
291  const Triangle& tOpo = triangles[iTopo];
292  const VertInd iVopo = opposedVertex(tOpo, iT);
293  const Vertex<T> vOpo = vertices[iVopo];
294 
295  intersected.push_back(iTopo);
296  iT = iTopo;
297  t = triangles[iT];
298 
299  PtLineLocation::Enum loc = locatePointLine(vOpo.pos, a.pos, b.pos);
300  if(loc == PtLineLocation::Left)
301  {
302  ptsLeft.push_back(iVopo);
303  iV = iVleft;
304  iVleft = iVopo;
305  }
306  else if(loc == PtLineLocation::Right)
307  {
308  ptsRight.push_back(iVopo);
309  iV = iVright;
310  iVright = iVopo;
311  }
312  else // encountered point on the edge
313  iB = iVopo;
314  }
315  // Remove intersected triangles
316  typedef std::vector<TriInd>::const_iterator TriIndCit;
317  for(TriIndCit it = intersected.begin(); it != intersected.end(); ++it)
318  makeDummy(*it);
319  // Triangulate pseudo-polygons on both sides
320  const TriInd iTleft = triangulatePseudopolygon(iA, iB, ptsLeft);
321  std::reverse(ptsRight.begin(), ptsRight.end());
322  const TriInd iTright = triangulatePseudopolygon(iB, iA, ptsRight);
323  changeNeighbor(iTleft, noNeighbor, iTright);
324  changeNeighbor(iTright, noNeighbor, iTleft);
325  // add fixed edge
326  fixedEdges.insert(Edge(iA, iB));
327  if(iB != edge.v2()) // encountered point on the edge
328  return insertEdge(Edge(iB, edge.v2()));
329 }

341 template <typename T>
342 tuple<TriInd, VertInd, VertInd> Triangulation<T>::intersectedTriangle(
343  const VertInd iA,
344  const std::vector<TriInd>& candidates,
345  const V2d<T>& a,
346  const V2d<T>& b) const
347 {
348  typedef std::vector<TriInd>::const_iterator TriIndCit;
349  for(TriIndCit it = candidates.begin(); it != candidates.end(); ++it)
350  {
351  const TriInd iT = *it;
352  const Triangle t = triangles[iT];
353  const Index i = vertexInd(t, iA);
354  const VertInd iP1 = t.vertices[cw(i)];
355  const VertInd iP2 = t.vertices[ccw(i)];
356  const PtLineLocation::Enum locP1 =
357  locatePointLine(vertices[iP1].pos, a, b);
358  const PtLineLocation::Enum locP2 =
359  locatePointLine(vertices[iP2].pos, a, b);
360  if(locP2 == PtLineLocation::Right)
361  {
362  if(locP1 == PtLineLocation::OnLine)
363  return make_tuple(noNeighbor, iP1, iP2);
364  if(locP1 == PtLineLocation::Left)
365  return make_tuple(iT, iP1, iP2);
366  }
367  }
368  throw std::runtime_error("Could not find vertex triangle intersected by "
369  "edge. Note: can be caused by duplicate points.");
370 }

372 template <typename T>
373 void Triangulation<T>::addSuperTriangle(const Box2d<T>& box)
374 {
375  const V2d<T> center = {(box.min.x + box.max.x) / T(2),
376  (box.min.y + box.max.y) / T(2)};
377  const T w = box.max.x - box.min.x;
378  const T h = box.max.y - box.min.y;
379  T r = std::sqrt(w * w + h * h) / 2.0; // incircle radius
380  r *= 1.1;
381  const T R = 2.0 * r; // excircle radius
382  const T shiftX = R * std::sqrt(T(3)) / 2.0; // R * cos(30 deg)
383  const V2d<T> posV1 = {center.x - shiftX, center.y - r};
384  const V2d<T> posV2 = {center.x + shiftX, center.y - r};
385  const V2d<T> posV3 = {center.x, center.y + R};
386  vertices.push_back(Vertex<T>::make(posV1, TriInd(0)));
387  vertices.push_back(Vertex<T>::make(posV2, TriInd(0)));
388  vertices.push_back(Vertex<T>::make(posV3, TriInd(0)));
389  const Triangle superTri = {{VertInd(0), VertInd(1), VertInd(2)},
390  {noNeighbor, noNeighbor, noNeighbor}};
391  addTriangle(superTri);
392 #ifdef CDT_USE_BOOST
393  if(m_closestPtMode == FindingClosestPoint::BoostRTree)
394  {
395  m_rtree.addPoint(posV1, VertInd(0));
396  m_rtree.addPoint(posV2, VertInd(1));
397  m_rtree.addPoint(posV3, VertInd(2));
398  }
399 #endif
400 }

402 template <typename T>
403 void Triangulation<T>::insertVertex(const V2d<T>& pos)
404 {
405  const VertInd iVert(vertices.size());
406  array<TriInd, 2> trisAt = walkingSearchTrianglesAt(pos);
407  std::stack<TriInd> triStack =
408  trisAt[1] == noNeighbor ? insertPointInTriangle(pos, trisAt[0])
409  : insertPointOnEdge(pos, trisAt[0], trisAt[1]);
410  while(!triStack.empty())
411  {
412  const TriInd iT = triStack.top();
413  triStack.pop();
414 
415  const Triangle& t = triangles[iT];
416  const TriInd iTopo = opposedTriangle(t, iVert);
417  if(iTopo == noNeighbor)
418  continue;
419  if(isFlipNeeded(pos, iT, iTopo, iVert))
420  {
421  flipEdge(iT, iTopo);
422  triStack.push(iT);
423  triStack.push(iTopo);
424  }
425  }
426 #ifdef CDT_USE_BOOST
427  if(m_closestPtMode == FindingClosestPoint::BoostRTree)
428  m_rtree.addPoint(pos, iVert);
429 #endif
430 }

442 template <typename T>
443 bool Triangulation<T>::isFlipNeeded(
444  const V2d<T>& pos,
445  const TriInd iT,
446  const TriInd iTopo,
447  const VertInd iVert) const
448 {
449  const Triangle& tOpo = triangles[iTopo];
450  const Index i = opposedVertexInd(tOpo, iT);
451  const VertInd iVopo = tOpo.vertices[i];
452  if(iVert < 3 && iVopo < 3) // opposed vertices belong to super-triangle
453  return false; // no flip is needed
454  const VertInd iVcw = tOpo.vertices[cw(i)];
455  const VertInd iVccw = tOpo.vertices[ccw(i)];
456  const V2d<T>& v1 = vertices[iVcw].pos;
457  const V2d<T>& v2 = vertices[iVopo].pos;
458  const V2d<T>& v3 = vertices[iVccw].pos;
459  if(iVcw < 3)
460  return locatePointLine(v1, v2, v3) == locatePointLine(pos, v2, v3);
461  if(iVccw < 3)
462  return locatePointLine(v3, v1, v2) == locatePointLine(pos, v1, v2);
463  return isInCircumcircle(pos, v1, v2, v3);
464 }

466 /* Insert point into triangle: split into 3 triangles:
467  * - create 2 new triangles
468  * - re-use old triangle for the 3rd
469  * v3
470  * / | \
471  * / | \ <-- original triangle (t)
472  * / | \
473  * n3 / | \ n2
474  * /newT2|newT1\
475  * / v \
476  * / __/ \__ \
477  * / __/ \__ \
478  * / _/ t' \_ \
479  * v1 ___________________ v2
480  * n1
481  */
482 template <typename T>
483 std::stack<TriInd>
484 Triangulation<T>::insertPointInTriangle(const V2d<T>& pos, const TriInd iT)
485 {
486  const VertInd v(vertices.size());
487  const TriInd iNewT1 = addTriangle();
488  const TriInd iNewT2 = addTriangle();
489 
490  Triangle& t = triangles[iT];
491  const array<VertInd, 3> vv = t.vertices;
492  const array<TriInd, 3> nn = t.neighbors;
493  const VertInd v1 = vv[0], v2 = vv[1], v3 = vv[2];
494  const TriInd n1 = nn[0], n2 = nn[1], n3 = nn[2];
495  // make two new triangles and convert current triangle to 3rd new triangle
496  using detail::arr3;
497  triangles[iNewT1] = Triangle::make(arr3(v2, v3, v), arr3(n2, iNewT2, iT));
498  triangles[iNewT2] = Triangle::make(arr3(v3, v1, v), arr3(n3, iT, iNewT1));
499  t = Triangle::make(arr3(v1, v2, v), arr3(n1, iNewT1, iNewT2));
500  // make and add a new vertex
501  vertices.push_back(Vertex<T>::makeInTriangle(pos, iT, iNewT1, iNewT2));
502  // adjust lists of adjacent triangles for v1, v2, v3
503  addAdjacentTriangle(v1, iNewT2);
504  addAdjacentTriangle(v2, iNewT1);
505  removeAdjacentTriangle(v3, iT);
506  addAdjacentTriangle(v3, iNewT1);
507  addAdjacentTriangle(v3, iNewT2);
508  // change triangle neighbor's neighbors to new triangles
509  changeNeighbor(n2, iT, iNewT1);
510  changeNeighbor(n3, iT, iNewT2);
511  // return newly added triangles
512  std::stack<TriInd> newTriangles;
513  newTriangles.push(iT);
514  newTriangles.push(iNewT1);
515  newTriangles.push(iNewT2);
516  return newTriangles;
517 }

519 /* Inserting a point on the edge between two triangles
520  * T1 (top) v1
521  * /|\
522  * n1 / | \ n4
523  * / | \
524  * / T1' | Tnew1\
525  * v2-------v-------v4
526  * \ Tnew2| T2' /
527  * \ | /
528  * n2 \ | / n3
529  * \|/
530  * T2 (bottom) v3
531  */
532 template <typename T>
533 std::stack<TriInd> Triangulation<T>::insertPointOnEdge(
534  const V2d<T>& pos,
535  const TriInd iT1,
536  const TriInd iT2)
537 {
538  const VertInd v(vertices.size());
539  const TriInd iTnew1 = addTriangle();
540  const TriInd iTnew2 = addTriangle();
541 
542  Triangle& t1 = triangles[iT1];
543  Triangle& t2 = triangles[iT2];
544  Index i = opposedVertexInd(t1, iT2);
545  const VertInd v1 = t1.vertices[i];
546  const VertInd v2 = t1.vertices[ccw(i)];
547  const TriInd n1 = t1.neighbors[i];
548  const TriInd n4 = t1.neighbors[cw(i)];
549  i = opposedVertexInd(t2, iT1);
550  const VertInd v3 = t2.vertices[i];
551  const VertInd v4 = t2.vertices[ccw(i)];
552  const TriInd n3 = t2.neighbors[i];
553  const TriInd n2 = t2.neighbors[cw(i)];
554  // add new triangles and change existing ones
555  using detail::arr3;
556  t1 = Triangle::make(arr3(v1, v2, v), arr3(n1, iTnew2, iTnew1));
557  t2 = Triangle::make(arr3(v3, v4, v), arr3(n3, iTnew1, iTnew2));
558  triangles[iTnew1] = Triangle::make(arr3(v1, v, v4), arr3(iT1, iT2, n4));
559  triangles[iTnew2] = Triangle::make(arr3(v3, v, v2), arr3(iT2, iT1, n2));
560  // make and add new vertex
561  vertices.push_back(Vertex<T>::makeOnEdge(pos, iT1, iTnew2, iT2, iTnew1));
562  // adjust neighboring triangles and vertices
563  changeNeighbor(n4, iT1, iTnew1);
564  changeNeighbor(n2, iT2, iTnew2);
565  addAdjacentTriangle(v1, iTnew1);
566  addAdjacentTriangle(v3, iTnew2);
567  removeAdjacentTriangle(v2, iT2);
568  addAdjacentTriangle(v2, iTnew2);
569  removeAdjacentTriangle(v4, iT1);
570  addAdjacentTriangle(v4, iTnew1);
571  // return newly added triangles
572  std::stack<TriInd> newTriangles;
573  newTriangles.push(iT1);
574  newTriangles.push(iTnew2);
575  newTriangles.push(iT2);
576  newTriangles.push(iTnew1);
577  return newTriangles;
578 }

580 template <typename T>
581 array<TriInd, 2> Triangulation<T>::trianglesAt(const V2d<T>& pos) const
582 {
583  array<TriInd, 2> out = {noNeighbor, noNeighbor};
584  for(TriInd i = TriInd(0); i < TriInd(triangles.size()); ++i)
585  {
586  const Triangle& t = triangles[i];
587  const V2d<T> v1 = vertices[t.vertices[0]].pos;
588  const V2d<T> v2 = vertices[t.vertices[1]].pos;
589  const V2d<T> v3 = vertices[t.vertices[2]].pos;
590  const PtTriLocation::Enum loc = locatePointTriangle(pos, v1, v2, v3);
591  if(loc == PtTriLocation::Outside)
592  continue;
593  out[0] = i;
594  if(isOnEdge(loc))
595  out[1] = t.neighbors[edgeNeighbor(loc)];
596  return out;
597  }
598  throw std::runtime_error("No triangle was found at position");
599 }

601 template <typename T>
602 TriInd Triangulation<T>::walkTriangles(
603  const VertInd startVertex,
604  const V2d<T>& pos) const
605 {
606  // begin walk in search of triangle at pos
607  TriInd currTri = vertices[startVertex].triangles[0];
608 #ifdef CDT_USE_BOOST
609  TriIndFlatUSet visited;
610 #else
611  TriIndUSet visited;
612 #endif
613  bool found = false;
614  while(!found)
615  {
616  const Triangle& t = triangles[currTri];
617  found = true;
618  // stochastic offset to randomize which edge we check first
619  const Index offset(detail::randGen() % 3);
620  for(Index i_(0); i_ < Index(3); ++i_)
621  {
622  const Index i((i_ + offset) % 3);

678 VertInd Triangulation<T>::nearestVertexRand(
679  const V2d<T>& pos,
680  const std::size_t nSamples) const
681 {
682  // start search at a vertex close to pos based on random sampling
683  VertInd out(detail::randGen() % vertices.size());

685  for(std::size_t iSample = 0; iSample < nSamples; ++iSample)

693  }

698 /* Flip edge between T and Topo:
699  *
700  * v4 | - old edge
701  * /|\ ~ - new edge
702  * / | \
703  * n3 / T' \ n4
704  * / | \
705  * / | \
706  * T -> v1~~~~~~~~~v3 <- Topo
707  * \ | /
708  * \ | /
709  * n1 \Topo'/ n2
710  * \ | /
711  * \|/
712  * v2
713  */
714 template <typename T>
715 void Triangulation<T>::flipEdge(const TriInd iT, const TriInd iTopo)
716 {
717  Triangle& t = triangles[iT];
718  Triangle& tOpo = triangles[iTopo];
719  const array<TriInd, 3>& triNs = t.neighbors;
720  const array<TriInd, 3>& triOpoNs = tOpo.neighbors;
721  const array<VertInd, 3>& triVs = t.vertices;
722  const array<VertInd, 3>& triOpoVs = tOpo.vertices;
723  // find vertices and neighbors
724  Index i = opposedVertexInd(t, iTopo);
725  const VertInd v1 = triVs[i];
726  const VertInd v2 = triVs[ccw(i)];
727  const TriInd n1 = triNs[i];
728  const TriInd n3 = triNs[cw(i)];
729  i = opposedVertexInd(tOpo, iT);
730  const VertInd v3 = triOpoVs[i];
731  const VertInd v4 = triOpoVs[ccw(i)];
732  const TriInd n4 = triOpoNs[i];
733  const TriInd n2 = triOpoNs[cw(i)];
734  // change vertices and neighbors
735  using detail::arr3;
736  t = Triangle::make(arr3(v4, v1, v3), arr3(n3, iTopo, n4));
737  tOpo = Triangle::make(arr3(v2, v3, v1), arr3(n2, iT, n1));
738  // adjust neighboring triangles and vertices
739  changeNeighbor(n1, iT, iTopo);
740  changeNeighbor(n4, iTopo, iT);
741  addAdjacentTriangle(v1, iTopo);
742  addAdjacentTriangle(v3, iT);
743  removeAdjacentTriangle(v2, iT);
744  removeAdjacentTriangle(v4, iTopo);
745 }

789 template <typename T>
790 TriInd Triangulation<T>::triangulatePseudopolygon(
791  const VertInd ia,
792  const VertInd ib,
793  const std::vector<VertInd>& points)
794 {
795  if(points.empty())
796  return pseudopolyOuterTriangle(ia, ib);
797  const VertInd ic = findDelaunayPoint(ia, ib, points);
798  const std::pair<std::vector<VertInd>, std::vector<VertInd> > splitted =
799  splitPseudopolygon(ic, points);
800  // triangulate splitted pseudo-polygons
801  TriInd iT2 = triangulatePseudopolygon(ic, ib, splitted.second);
802  TriInd iT1 = triangulatePseudopolygon(ia, ic, splitted.first);
803  // add new triangle
804  const Triangle t = {{ia, ib, ic}, {noNeighbor, iT2, iT1}};
805  const TriInd iT = addTriangle(t);
806  // adjust neighboring triangles and vertices
807  if(iT1 != noNeighbor)
808  {
809  if(splitted.first.empty())
810  changeNeighbor(iT1, ia, ic, iT);
811  else
812  triangles[iT1].neighbors[0] = iT;
813  }
814  if(iT2 != noNeighbor)
815  {
816  if(splitted.second.empty())
817  changeNeighbor(iT2, ic, ib, iT);
818  else
819  triangles[iT2].neighbors[0] = iT;
820  }
821  addAdjacentTriangle(ia, iT);
822  addAdjacentTriangle(ib, iT);
823  addAdjacentTriangle(ic, iT);
824 
825  return iT;
826 }

831  const VertInd ib,

877 DuplicatesInfo RemoveDuplicates(std::vector<V2d<T> >& vertices)
878 {

883

889  uniqueVerts.insert(std::make_pair(vertices[iIn], iOut));

901  remove_at(
902  vertices.begin(),
903  vertices.end(),
904  removedDuplicateIndices.begin(),
905  removedDuplicateIndices.end()),
906  vertices.end());

909  return duplicateVertices;

932 TriIndUSet PeelLayer(
933  std::stack<TriInd> seeds,
934  const TriangleVec& triangles,
935  const EdgeUSet& fixedEdges,
936  const unsigned short layerDepth,
937  std::vector<unsigned short>& triDepths)
938 {

940  while(!seeds.empty())
941  {

943  seeds.pop();
944  triDepths[iT] = layerDepth;
945  behindBoundary.erase(iT);
